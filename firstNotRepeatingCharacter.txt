function firstNotRepeatingCharacter(s){
    var array=[];
    var array2=[];
    for(i=0;i<s.length;i++){
        array2.push(s.charAt(i));
        for(x=i+1;x<s.length;x++){
            if(s.charAt(i)==s.charAt(x)){
                array.push(s.charAt(i));
            }
        }
        for(x=0;x<array2.length;x++){
            if(array[i]==array2[x]){
                array2.splice(x,1);                
            }
        }
    }
    return array2[0];    
}
